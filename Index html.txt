<script>
function getData(){
fetch("https://jsonplaceholder.typicode.com/posts")
.then(res => res.json())
.then(data =>{
let output = "";
data.slice(0,10).forEach(post =>{
output += `
<div class="card">
<h3>${post.title}</h3>
<p>${post.body}</p>
</div>
`;
});
document.getElementById("result").innerHTML = output;
})
.catch(err =>{
document.getElementById("result").innerHTML = "<p>Error loading data</p>";
});
}
</script>
